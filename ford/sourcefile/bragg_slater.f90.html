  subroutine set_bragg_slater(array, bstype)
    real(kind=dp), intent(out) :: array(:)
    integer, intent(in) :: bstype

    select case (bstype)
    case (BRSL_TYPE_GAMESS)
      array = brsl_values_gamess

    case (BRSL_TYPE_TA)
      array = brsl_values_treutler

    case (BRSL_TYPE_GILL)

    case default
      array = brsl_values_gill
