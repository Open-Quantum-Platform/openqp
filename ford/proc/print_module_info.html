  subroutine print_module_info(module_title,module_info)
    use io_constants, only: iw

    implicit none

    character(len=*), intent(in) :: module_title, module_info

    write(iw,'(/20x,40("+")/&
             &23X,"MODULE: ",A/&
             &23X,A/&
             &20X,40("+"))') module_title, module_info

  end subroutine print_module_info
