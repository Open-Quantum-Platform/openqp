  subroutine get_transitions(trans, noca, nocb, nbf)

    implicit none

    integer, intent(out), dimension(:,:) :: trans
    integer, intent(in) :: noca, nocb, nbf

    integer :: ij, i, j
    ij = 0
    do j = nocb+1, nbf
       do i = 1, noca
        ij = ij+1
        trans(ij,1) = i
        trans(ij,2) = j
      end do
    end do

  end subroutine get_transitions
