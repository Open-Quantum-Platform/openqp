  subroutine pcgb(pk,r,xm_in)
    use precision, only: dp

    implicit none

    real(kind=dp), intent(out) :: pk(:)
    real(kind=dp), intent(in) :: r(:)
    real(kind=dp), intent(in) :: xm_in(:)

    real(kind=dp) :: beta

    beta = 1.0_dp/sum(r*r*xm_in)

  ! pk ini
    pk = pk + beta*xm_in*r

  end subroutine pcgb
